\section{Integriranje}
\s{Kvadraturne forme}\\
\textbf{Kvadraturna formula:} \(\int_{a}^{b}f(x)\, dx = \sum_{i=0}^{n}w_i f(x_i) + R(f)\), kjer paroma različni točki \(x_0, \ldots, x_n\) so \emph{vozli}, \(w_i = \int_{a}^{b}l_{n,i}(x) \, dx\) so \emph{uteži}, \(R(f) = \int_{a}^{b} \frac{f^{(n+1)}(\xi_x)}{(n+1)!} \omega(x) \, dx\) je napaka metode.\\
\s{Newton-Cotesove formule.} Vozle (\(n\)) izberimo ekvidistantno (\(h\)).
\begin{itemize}
  \item \emph{Zaprti tip}: upoštevamo krajišča integracije;
  \begin{itemize}
    \item \(n=1: \int_{x_0}^{x_1} f(x)\, dx = \frac{h}{2}(f_0 + f_1) - \frac{h^3}{12}f''(\xi)\)
    \item \(n=2: \int_{x_0}^{x_2} f(x) \, dx = \frac{h}{3}(f_0+4f_1+f_2) - \frac{h^5}{90}f^{(4)}(\xi)\)
    \item \(n=3: \int_{x_0}^{x_3} f(x) \, dx = \frac{3h}{8}(f_0 + 3 f_1 + 3 f_2 + f_3) - \frac{3h^5}{80}f^{(4)}(\xi)\)
  \end{itemize}
  \item \emph{Odpri tip}: ni krajišč.
  \begin{itemize}
    \item \(n=2: \int_{x_0}^{x_2} f(x) \, dx = 2hf_1 + \frac{h^3}{3}f''(\xi)\)
    \item \(n=4: \int_{x_0}^{x_4} f(x) \, dx = \frac{4h}{3}(2f_1-f_2+2f_3) + \frac{28h^5}{90}f^{(4)}(\xi)\)
  \end{itemize}
\end{itemize}
\s{Sestavljena pravila.} Interval razdelimo na podintervale. Na vsakem podintervalu uporabimo kvadraturno formulo nizke stopnje.\\
